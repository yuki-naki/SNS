create table user_t(
	user_id varchar2(12) primary key,
	user_name varchar2(30),
	user_icon blob,
	user_login_id varchar2(20),
	user_password varchar2(20),
	user_introduce varchar2(500),
	student_id number(12),
	student_admission_year varchar2(4),
	grouping varchar2(1)
)

create table group_t(
	group_id varchar2(12) primary key,
	group_name varchar2(100)
)

create table groupmember_t(
	group_id varchar2(12),
	user_id varchar2(100),
	foreign key (group_id) references group_t(group_id),
	foreign key (user_id) references user_t(user_id)
)

create table department_t(
	Department_id varchar2(12) primary key,
	Department_name varchar2(30) not null
)	

create table belong_department_t(
	user_id varchar2(12) references user_t(user_id),
	department_id varchar2(12) references department_t(department_id)
)

create table follow_t(
	follower_user_id varchar2(12) references user_t(user_id),
	followed_user_id varchar2(12) references user_t(user_id)
)

create table contribution_t(
	contribution_id varchar2(12) primary key,
	contribution_content varchar2(1000),
	contribution_date date,
	user_id varchar2(12) references user_t(user_id)
)

create table chat_t(
	user_id varchar2(12) references user_t(user_id),
	group_id varchar2(12) references group_t(group_id),
	chat_content varchar2(600),
	chat_date date
)
